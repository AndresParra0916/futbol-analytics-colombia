def extraer_jugadores_de_equipo(equipo):
    print(f"Extrayendo {equipo['nombre']}...")
    try:
        response = requests.get(equipo['url'], headers=HEADERS, timeout=30)
        response.raise_for_status()
        soup = BeautifulSoup(response.text, 'html.parser')
        
        # Buscar la tabla de jugadores (usualmente una tabla con clase 'wp-block-table')
        tabla = soup.find('table')
        if not tabla:
            print(f"  No se encontró tabla para {equipo['nombre']}")
            return []
        
        filas = tabla.find_all('tr')
        jugadores = []
        for fila in filas[1:]:  # saltar encabezado
            columnas = fila.find_all('td')
            if len(columnas) >= 6:
                nombre = columnas[0].get_text(strip=True)
                edad = columnas[1].get_text(strip=True)
                pie = columnas[2].get_text(strip=True)
                posicion = columnas[3].get_text(strip=True)
                numero = columnas[4].get_text(strip=True)
                jugadores.append({
                    'nombre': nombre,
                    'equipo': equipo['nombre'],
                    'edad': edad,
                    'pie_dominante': pie,
                    'posicion': posicion,
                    'numero': numero
                })
        print(f"  → {len(jugadores)} jugadores extraídos")
        time.sleep(1)  # pausa para no sobrecargar el servidor
        return jugadores
    except Exception as e:
        print(f"  Error con {equipo['nombre']}: {e}")
        return []
